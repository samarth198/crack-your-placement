class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int sum = 0;
        int currentSum = 0;
        int count = 0;
        unordered_map<int,int> prevSum;
        prevSum[sum]=1;
        for(int i = 0;i<nums.size();i++){
           currentSum += nums[i];
        
            if (prevSum.find(currentSum - k) != prevSum.end()) {
            count += prevSum[currentSum - k];
            }
        
      
            prevSum[currentSum]++;
        }
    
    return count;
    }
};
