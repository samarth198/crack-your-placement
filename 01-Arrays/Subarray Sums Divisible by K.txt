class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        unordered_map<int,int> map;
        int sum = 0;
        int count = 0;
        int mod;
        map[sum]=1;
        for(int i =0;i<nums.size();i++){
            sum+=nums[i];
            mod = sum%k;
            if(mod < 0) mod+=k;
            if(map.find(mod) != map.end()) count+=map[mod];

            map[mod]++;
        }
    return count;
    }

};
