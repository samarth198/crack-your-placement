class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        
        int total = 0;
        for (int i = n - k; i < n; ++i) {
            total += cardPoints[i];
        }
        
        int maxScore = total;
        int currentScore = total;
        
            for (int i = 0; i < k; ++i) {
            currentScore += cardPoints[i] - cardPoints[n - k + i];
            maxScore = max(maxScore, currentScore);
        }
        
        return maxScore;
    }
};
