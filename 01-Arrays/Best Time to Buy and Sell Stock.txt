class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int maxProfit = 0;
        int minimum = INT_MAX;
        for(int i = 0;i<prices.size();i++){
            minimum = min(prices[i],minimum);
            maxProfit = max(maxProfit,prices[i]-minimum);
        }
        return maxProfit;
    }
};
