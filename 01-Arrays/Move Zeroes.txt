class Solution {
public:
    void moveZeroes(vector<int>& nums) {
       int p1,p2;
       for(p1 = 0;p1<nums.size();p1++){
        if(nums[p1]==0) break;
       }
       cout << p1;
       for(p2=p1+1;p2<nums.size();p2++){
        if(nums[p2]!=0){
            swap(nums[p2],nums[p1]);
            p1++;
        }
       }
    }
};
