class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode* temp = head;
        int count = 0;
        while(temp!=NULL){
            count++;
            temp = temp->next;
        }
        if(count%2 == 0) count = count/2 + 1;
        else count = (count+1)/2;

        while(--count){
            head = head->next;
        }
        return head;
    }
};
